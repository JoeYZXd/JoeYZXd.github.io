# 切换python virtualenv
switch-to() {
  if [ ! -n "$1" ]; then
    echo "请输入要切换的虚拟环境名称"
  else
    envPath="/home/joey/$1/bin/activate"
    if [ ! -e "$envPath" ]; then
      echo "虚拟环境[$1]不存在"
    else
      echo "切换到虚拟环境[$1]"
      source $envPath
    fi
  fi
}

alias swn="switch-to nikola"
alias sww="switch-to web"
